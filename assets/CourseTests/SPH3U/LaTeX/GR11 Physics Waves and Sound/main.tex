\section*{Part B --- Short Answer (33 marks)}

\question A wave on a string has a frequency of \SI{5.0}{Hz} and a wavelength of
\SI{0.60}{m}.
\begin{parts}
    \part[2] Calculate the speed of the wave.
    \part[1] Calculate the period of the wave.
    \part[2] The wave enters a new medium and slows to \SI{2.0}{m/s}. What is the
    new wavelength? (Frequency does not change at a boundary.)
    \part[2] How many complete wavelengths fit in a \SI{4.5}{m} string vibrating at
    the original conditions?
\end{parts}
\begin{solution}[8cm]
\textbf{(a)} $v = f\lambda = 5.0 \times 0.60 = \mathbf{\SI{3.0}{m/s}}$

\textbf{(b)} $T = \dfrac{1}{f} = \dfrac{1}{5.0} = \mathbf{\SI{0.20}{s}}$

\textbf{(c)} $\lambda' = \dfrac{v'}{f} = \dfrac{2.0}{5.0} = \mathbf{\SI{0.40}{m}}$

\textbf{(d)} $n = \dfrac{L}{\lambda} = \dfrac{4.5}{0.60} = \mathbf{7.5}$ wavelengths
\end{solution}

\question Resonance in air columns. The speed of sound is \SI{343}{m/s}.
\begin{parts}
    \part[3] An \textbf{open} air column (open at both ends) resonates at its fundamental
    frequency with a length of \SI{0.65}{m}. Find the fundamental frequency.
    \part[3] A \textbf{closed} air column (closed at one end) has a length of
    \SI{0.45}{m}. Find its fundamental frequency and its first overtone frequency.
    \part[2] A closed pipe resonates at \SI{572}{Hz}. What is its length?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} Open column, fundamental: $L = \dfrac{\lambda}{2} \Rightarrow \lambda = 2L = \SI{1.30}{m}$
\[
f = \frac{v}{\lambda} = \frac{343}{1.30} = \mathbf{\SI{264}{Hz}}
\]

\textbf{(b)} Closed column, fundamental: $L = \dfrac{\lambda}{4} \Rightarrow \lambda = 4L = \SI{1.80}{m}$
\[
f_1 = \frac{343}{1.80} = \mathbf{\SI{190.6}{Hz}}
\]
Closed columns support only odd harmonics; first overtone = 3rd harmonic:
\[
f_3 = 3 f_1 = 3 \times 190.6 = \mathbf{\SI{571.8}{Hz} \approx \SI{572}{Hz}}
\]

\textbf{(c)} Closed column fundamental: $L = \dfrac{v}{4f} = \dfrac{343}{4 \times 572}
= \dfrac{343}{2288} = \mathbf{\SI{0.150}{m}}$
\end{solution}

\question The Doppler Effect.
An ambulance emits a siren at \SI{900}{Hz}. The speed of sound is \SI{343}{m/s}.
\begin{parts}
    \part[2] The ambulance approaches a stationary observer at \SI{30}{m/s}. What
    frequency does the observer hear?
    \part[2] The ambulance moves away from the observer at \SI{30}{m/s}. What frequency
    is heard?
    \part[3] The observer is now in a car moving \textbf{toward} the stationary ambulance
    at \SI{20}{m/s}. What frequency does the observer hear?
\end{parts}
\begin{solution}[9cm]
Doppler formula: $f' = f_s \cdot \dfrac{v \pm v_o}{v \mp v_s}$
(upper signs when approaching, lower when receding)

\textbf{(a)} Source approaching, observer stationary ($v_o = 0$):
\[
f' = 900 \times \frac{343}{343 - 30} = 900 \times \frac{343}{313} = \mathbf{\SI{986}{Hz}}
\]

\textbf{(b)} Source receding:
\[
f' = 900 \times \frac{343}{343 + 30} = 900 \times \frac{343}{373} = \mathbf{\SI{827}{Hz}}
\]

\textbf{(c)} Observer approaching stationary source ($v_s = 0$):
\[
f' = 900 \times \frac{343 + 20}{343} = 900 \times \frac{363}{343} = \mathbf{\SI{953}{Hz}}
\]
\end{solution}

\question Interference and standing waves.
\begin{parts}
    \part[2] Explain the difference between constructive and destructive interference.
    Under what conditions does each occur?
    \part[3] A guitar string of length \SI{0.65}{m} vibrates in its third harmonic.
    Draw the standing wave pattern, state the wavelength, and find the frequency
    if the wave speed on the string is \SI{400}{m/s}.
    \part[2] Two speakers emit the same frequency. A listener stands \SI{3.0}{m}
    from one speaker and \SI{3.5}{m} from the other. If the speed of sound is
    \SI{343}{m/s} and they hear a minimum (destructive interference), find the
    lowest frequency that produces this minimum.
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} \textbf{Constructive interference} occurs when two waves meet in phase
(path difference $= n\lambda$, $n = 0,1,2,\ldots$); their amplitudes add. \textbf{Destructive
interference} occurs when they meet out of phase (path difference $= (n+\tfrac{1}{2})\lambda$);
their amplitudes cancel.

\textbf{(b)} A string fixed at both ends: $n$th harmonic has $n$ antinodes.
3rd harmonic: 3 loops, $L = \tfrac{3\lambda}{2} \Rightarrow \lambda = \dfrac{2L}{3}
= \dfrac{2(0.65)}{3} = \mathbf{\SI{0.433}{m}}$
\[
f = \frac{v}{\lambda} = \frac{400}{0.433} = \mathbf{\SI{923}{Hz}}
\]

\textbf{(c)} Path difference $= 3.5 - 3.0 = \SI{0.5}{m}$.
First minimum: path difference $= \dfrac{\lambda}{2} \Rightarrow \lambda = \SI{1.0}{m}$
\[
f = \frac{v}{\lambda} = \frac{343}{1.0} = \mathbf{\SI{343}{Hz}}
\]
\end{solution}
