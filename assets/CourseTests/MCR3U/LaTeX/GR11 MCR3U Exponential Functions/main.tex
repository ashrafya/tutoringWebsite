\section*{Part B --- Short Answer (33 marks)}

\question Simplify each expression. Write answers with positive exponents only.
\begin{parts}
    \part[2] $\dfrac{(3a^2b^{-1})^3}{9a^{-1}b^2}$
    \part[2] $\left(\dfrac{16x^4}{y^{-2}}\right)^{1/2} \cdot \dfrac{y}{2x}$
    \part[2] $\dfrac{5^{n+2} - 5^n}{5^n}$
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} Numerator: $27a^6 b^{-3}$
\[
\frac{27a^6 b^{-3}}{9a^{-1}b^2} = 3a^{6-(-1)}b^{-3-2} = 3a^7 b^{-5} = \mathbf{\frac{3a^7}{b^5}}
\]

\textbf{(b)} $\left(\dfrac{16x^4}{y^{-2}}\right)^{1/2} = \left(16x^4 y^2\right)^{1/2}
= 4x^2 y$
\[
4x^2 y \cdot \frac{y}{2x} = \frac{4x^2 y^2}{2x} = \mathbf{2xy^2}
\]

\textbf{(c)} $\dfrac{5^n(5^2 - 1)}{5^n} = 5^2 - 1 = \mathbf{24}$
\end{solution}

\question Solve each exponential equation. Show all steps using equal bases.
\begin{parts}
    \part[2] $3^{2x-1} = 81$
    \part[2] $4^{x+1} = 8^{2x-3}$
    \part[3] $5^{2x} - 6 \cdot 5^x + 5 = 0$\quad (Hint: let $u = 5^x$)
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $3^{2x-1} = 3^4 \Rightarrow 2x - 1 = 4 \Rightarrow x = \mathbf{\dfrac{5}{2}}$

\textbf{(b)} $2^{2(x+1)} = 2^{3(2x-3)} \Rightarrow 2x+2 = 6x-9 \Rightarrow 11 = 4x
\Rightarrow x = \mathbf{\dfrac{11}{4}}$

\textbf{(c)} Let $u = 5^x$: $u^2 - 6u + 5 = 0 \Rightarrow (u-1)(u-5) = 0$

$u = 1 \Rightarrow 5^x = 5^0 \Rightarrow \mathbf{x = 0}$;\quad
$u = 5 \Rightarrow 5^x = 5^1 \Rightarrow \mathbf{x = 1}$
\end{solution}

\question A bacterial culture starts with 500 bacteria and doubles every 3 hours.
\begin{parts}
    \part[2] Write an equation for the number of bacteria $N$ after $t$ hours.
    \part[2] How many bacteria are present after 12 hours?
    \part[3] After how many hours will there be 32\,000 bacteria?
\end{parts}
\begin{solution}[8cm]
\textbf{(a)} Each 3 hours: multiply by 2. $N(t) = 500 \cdot 2^{t/3}$\quad
$\mathbf{N(t) = 500 \cdot 2^{t/3}}$

\textbf{(b)} $N(12) = 500 \cdot 2^{12/3} = 500 \cdot 2^4 = 500 \times 16 = \mathbf{8000}$

\textbf{(c)} $500 \cdot 2^{t/3} = 32\,000 \Rightarrow 2^{t/3} = 64 = 2^6
\Rightarrow \dfrac{t}{3} = 6 \Rightarrow \mathbf{t = 18 \text{ hours}}$
\end{solution}

\question A radioactive isotope has a half-life of 20 years. A sample currently has
a mass of 80 g.
\begin{parts}
    \part[2] Write a function $M(t)$ for the mass remaining after $t$ years.
    \part[2] What mass remains after 60 years?
    \part[3] Sketch the graph of $M(t)$ for $0 \leq t \leq 100$. Label the asymptote,
    $y$-intercept, and at least two points.
    \part[2] Describe the transformations from $y = \left(\tfrac{1}{2}\right)^x$
    to $y = M(t)$.
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $M(t) = 80 \cdot \left(\dfrac{1}{2}\right)^{t/20}$

\textbf{(b)} $M(60) = 80 \cdot \left(\dfrac{1}{2}\right)^{3} = 80 \times \dfrac{1}{8}
= \mathbf{\SI{10}{g}}$

\textbf{(c)} Graph: decreasing exponential. $y$-intercept $(0, 80)$;
point $(20, 40)$; point $(40, 20)$; point $(60, 10)$.
Horizontal asymptote: $M = 0$.

\textbf{(d)} From $y = \left(\tfrac{1}{2}\right)^x$ to $y = 80\cdot\left(\tfrac{1}{2}\right)^{t/20}$:
horizontal stretch by factor 20 (period of halving changes from 1 to 20), then
vertical stretch by factor 80 (initial mass).
\end{solution}
