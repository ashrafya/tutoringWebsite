\question Solve each system of equations. Show all work. \textbf{(6 marks)}

\begin{parts}

\part[3] Solve by \textbf{substitution}: \quad $\begin{cases} y = 3x - 4 \\ 2x + 5y = 6 \end{cases}$

\begin{solution}[6cm]
Substitute $y = 3x - 4$ into the second equation:
\[
2x + 5(3x - 4) = 6
\]
\[
2x + 15x - 20 = 6
\]
\[
17x = 26
\]
\[
x = \frac{26}{17}
\]
\[
y = 3\!\left(\frac{26}{17}\right) - 4 = \frac{78}{17} - \frac{68}{17} = \frac{10}{17}
\]
Solution: $\left(\dfrac{26}{17},\ \dfrac{10}{17}\right)$
\end{solution}

\part[3] Solve by \textbf{elimination}: \quad $\begin{cases} 3x + 2y = 14 \\ 5x - 2y = 10 \end{cases}$

\begin{solution}[6cm]
Add the two equations to eliminate $y$:
\[
(3x + 2y) + (5x - 2y) = 14 + 10
\]
\[
8x = 24 \implies x = 3
\]
Substitute $x = 3$ into the first equation:
\[
3(3) + 2y = 14 \implies 9 + 2y = 14 \implies y = \frac{5}{2}
\]
Solution: $\left(3,\ \dfrac{5}{2}\right)$
\end{solution}

\end{parts}

\question \textbf{Cell Phone Plans. (5 marks)}\\
Plan A costs \$25 per month plus \$0.10 per text message.
Plan B costs \$15 per month plus \$0.20 per text message.
\begin{parts}
\part[2] Write an equation for the monthly cost of each plan in terms of $n$, the number of text messages.
\begin{solution}[3cm]
Plan A: $C_A = 0.10n + 25$\\
Plan B: $C_B = 0.20n + 15$
\end{solution}

\part[3] Find the number of text messages at which both plans cost the same. State which plan is cheaper for fewer messages.
\begin{solution}[5cm]
Set equal:
\[
0.10n + 25 = 0.20n + 15
\]
\[
10 = 0.10n \implies n = 100 \text{ messages}
\]
At $n = 100$, both plans cost \$35.

For fewer than 100 messages, Plan B is cheaper (lower fixed cost). For more than 100 messages, Plan A is cheaper.
\end{solution}
\end{parts}

\question \textbf{Applications. (6 marks)}
\begin{parts}
\part[3] A chemist needs 500~mL of a 40\% acid solution. She has a 25\% solution and a 70\% solution. How many millilitres of each should she mix?

\begin{solution}[7cm]
Let $x$ = mL of 25\% solution and $y$ = mL of 70\% solution.
\[
x + y = 500
\]
\[
0.25x + 0.70y = 0.40(500) = 200
\]
From the first equation: $x = 500 - y$. Substitute:
\[
0.25(500 - y) + 0.70y = 200
\]
\[
125 - 0.25y + 0.70y = 200
\]
\[
0.45y = 75 \implies y = \frac{500}{3} \approx 166.7 \text{ mL of 70\% solution}
\]
\[
x = 500 - \frac{500}{3} = \frac{1000}{3} \approx 333.3 \text{ mL of 25\% solution}
\]
\end{solution}

\part[3] Two trains leave different stations 420~km apart and travel toward each other. Train A travels at 90~km/h and Train B at 60~km/h. How long until they meet, and how far has each train travelled?

\begin{solution}[7cm]
Let $t$ = time (hours) until meeting.
\[
90t + 60t = 420
\]
\[
150t = 420 \implies t = 2.8 \text{ hours}
\]
Distance by Train A: $90 \times 2.8 = 252$ km\\
Distance by Train B: $60 \times 2.8 = 168$ km\\
Check: $252 + 168 = 420$. \checkmark
\end{solution}
\end{parts}

\question \textbf{System Types and Graphing. (8 marks)}
\begin{parts}
\part[3] Without solving, identify whether each system has one solution, no solution, or infinitely many solutions. Justify your answer.
\begin{enumerate}[(i)]
    \item $\begin{cases} 4x - 2y = 6 \\ 2x - y = 3 \end{cases}$
    \item $\begin{cases} 3x + y = 5 \\ 3x + y = 8 \end{cases}$
    \item $\begin{cases} x + 2y = 4 \\ 3x - y = 5 \end{cases}$
\end{enumerate}

\begin{solution}[6cm]
\begin{enumerate}[(i)]
    \item The first equation is $2(2x - y) = 6$, i.e., $2x - y = 3$, which is the same as the second. \textbf{Infinitely many solutions} (same line).
    \item Same left side but different right sides. Lines are parallel and distinct. \textbf{No solution.}
    \item Different slopes: slope from (i) is $-\tfrac{1}{2}$, from (ii) is $3$. Lines intersect. \textbf{One solution.}
\end{enumerate}
\end{solution}

\part[3] Graph the system from (iii) above ($x + 2y = 4$ and $3x - y = 5$) on the grid below and label the solution point.

\begin{solution}[7cm]
Line 1: $x + 2y = 4 \Rightarrow y = -\tfrac{1}{2}x + 2$. Passes through $(0, 2)$ and $(4, 0)$.\\
Line 2: $3x - y = 5 \Rightarrow y = 3x - 5$. Passes through $(0, -5)$ and $(\tfrac{5}{3}, 0)$.\\
Solve: $-\tfrac{1}{2}x + 2 = 3x - 5 \Rightarrow 7 = \tfrac{7}{2}x \Rightarrow x = 2,\ y = 1$.\\
\textbf{Solution: $(2, 1)$}
\end{solution}

\part[2] Verify that your solution to part (iii) is correct by substituting into both original equations.

\begin{solution}[3cm]
$(2, 1)$:\\
$x + 2y = 2 + 2(1) = 4$ \checkmark\\
$3x - y = 3(2) - 1 = 5$ \checkmark
\end{solution}
\end{parts}

\question \textbf{Geometry Context. (8 marks)}
\begin{parts}
\part[3] The perimeter of a rectangle is 56~cm. The length is 4~cm more than twice the width. Find the dimensions of the rectangle.

\begin{solution}[6cm]
Let $l$ = length and $w$ = width.
\[
2l + 2w = 56 \implies l + w = 28
\]
\[
l = 2w + 4
\]
Substitute: $(2w + 4) + w = 28 \implies 3w = 24 \implies w = 8$ cm, $l = 20$ cm.

Check: Perimeter $= 2(20) + 2(8) = 56$ cm. \checkmark
\end{solution}

\part[2] Two angles are supplementary. One angle is $15^\circ$ less than three times the other. Find both angles.

\begin{solution}[5cm]
Let the angles be $\alpha$ and $\beta$.
\[
\alpha + \beta = 180^\circ
\]
\[
\alpha = 3\beta - 15^\circ
\]
Substitute: $(3\beta - 15) + \beta = 180 \implies 4\beta = 195 \implies \beta = 48.75^\circ$, $\alpha = 131.25^\circ$.
\end{solution}

\part[3] A triangle has a perimeter of 45~cm. The longest side is twice the shortest side. The third side is 9~cm more than the shortest side. Find all three side lengths.

\begin{solution}[6cm]
Let $s$ = shortest side.\\
Longest side $= 2s$. Third side $= s + 9$.
\[
s + 2s + (s + 9) = 45
\]
\[
4s + 9 = 45 \implies 4s = 36 \implies s = 9 \text{ cm}
\]
Sides: 9~cm, 18~cm, 18~cm. (The triangle is isosceles.)

Check: $9 + 18 + 18 = 45$ cm. \checkmark
\end{solution}
\end{parts}
