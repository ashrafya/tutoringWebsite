\question \textbf{Expand and Simplify. (6 marks)}\\
Expand and simplify each expression.

\begin{parts}
\part[2] $(2x - 5)(3x + 4)$
\begin{solution}[4cm]
\[
(2x - 5)(3x + 4) = 6x^2 + 8x - 15x - 20 = 6x^2 - 7x - 20
\]
\end{solution}

\part[2] $(x + 3)^2 - (x - 1)(x + 1)$
\begin{solution}[5cm]
\[
(x + 3)^2 = x^2 + 6x + 9
\]
\[
(x-1)(x+1) = x^2 - 1
\]
\[
(x^2 + 6x + 9) - (x^2 - 1) = 6x + 10
\]
\end{solution}

\part[2] $(4x - 1)(4x + 1) + (2x - 3)^2$
\begin{solution}[5cm]
\[
(4x-1)(4x+1) = 16x^2 - 1
\]
\[
(2x-3)^2 = 4x^2 - 12x + 9
\]
\[
16x^2 - 1 + 4x^2 - 12x + 9 = 20x^2 - 12x + 8
\]
\end{solution}
\end{parts}

\question \textbf{Factor Completely. (6 marks)}\\
Factor each expression completely.

\begin{parts}
\part[1] $15x^3y - 20x^2y^2$ \quad (common factor)
\begin{solution}[3cm]
$= 5x^2y(3x - 4y)$
\end{solution}

\part[1] $49x^2 - 64$ \quad (difference of squares)
\begin{solution}[3cm]
$= (7x - 8)(7x + 8)$
\end{solution}

\part[2] $x^2 - 3x - 28$ \quad (trinomial, $a = 1$)
\begin{solution}[4cm]
Find two numbers that multiply to $-28$ and add to $-3$: $-7$ and $4$.
\[
= (x - 7)(x + 4)
\]
\end{solution}

\part[2] $6x^2 + 11x - 10$ \quad (trinomial, $a \neq 1$, decomposition)
\begin{solution}[5cm]
Product $= 6 \times (-10) = -60$. Need two numbers multiplying to $-60$ and adding to $11$: $15$ and $-4$.
\[
6x^2 + 15x - 4x - 10 = 3x(2x + 5) - 2(2x + 5) = (3x - 2)(2x + 5)
\]
\end{solution}
\end{parts}

%---
\question \textbf{Factor by Decomposition. (6 marks)}

\begin{parts}
\part[3] Factor completely: $4x^2 - 4x - 15$

\begin{solution}[6cm]
Product $= 4 \times (-15) = -60$. Need two numbers multiplying to $-60$ and adding to $-4$: $-10$ and $6$.
\[
4x^2 - 10x + 6x - 15 = 2x(2x - 5) + 3(2x - 5) = (2x + 3)(2x - 5)
\]
\end{solution}

\part[3] Factor completely: $6x^2 - 7x - 3$

\begin{solution}[6cm]
Product $= 6 \times (-3) = -18$. Need two numbers multiplying to $-18$ and adding to $-7$: $-9$ and $2$.
\[
6x^2 - 9x + 2x - 3 = 3x(2x - 3) + 1(2x - 3) = (3x + 1)(2x - 3)
\]
\end{solution}
\end{parts}

%---
\question \textbf{Mixed Factoring. (7 marks)}\\
Factor each expression completely. Some may require more than one step.

\begin{parts}
\part[1] $x^2 - 100$
\begin{solution}[2cm]
$(x-10)(x+10)$
\end{solution}

\part[2] $2x^2 - 8x - 42$
\begin{solution}[4cm]
$= 2(x^2 - 4x - 21) = 2(x - 7)(x + 3)$
\end{solution}

\part[2] $3x^3 - 12x$
\begin{solution}[4cm]
$= 3x(x^2 - 4) = 3x(x - 2)(x + 2)$
\end{solution}

\part[2] $9x^2 - 6x + 1$
\begin{solution}[4cm]
Recognise as a perfect square: $(3x - 1)^2$.
\end{solution}
\end{parts}

\question \textbf{Applications. (8 marks)}

\begin{parts}
\part[3] A rectangle has length $(3x + 2)$ cm and width $(x + 4)$ cm. Write an expression for the area of the rectangle and expand it fully.

\begin{solution}[5cm]
\[
A = (3x + 2)(x + 4) = 3x^2 + 12x + 2x + 8 = 3x^2 + 14x + 8
\]
\end{solution}

\part[3] A large square has side length $(2x + 5)$ cm. A smaller square of side length $x$ cm is cut from one corner. Write a simplified expression for the remaining area, then factor it completely.

\begin{solution}[6cm]
Large square area: $(2x+5)^2 = 4x^2 + 20x + 25$\\
Small square area: $x^2$\\
Remaining area: $4x^2 + 20x + 25 - x^2 = 3x^2 + 20x + 25$\\
Factor: need numbers multiplying to $3 \times 25 = 75$ and adding to $20$: $5$ and $15$.
\[
3x^2 + 5x + 15x + 25 = x(3x + 5) + 5(3x + 5) = (x + 5)(3x + 5)
\]
\end{solution}

\part[2] Write a simplified expression for the perimeter of the rectangle in part (a), then factor out the GCF.

\begin{solution}[4cm]
Perimeter $= 2(3x + 2) + 2(x + 4) = 6x + 4 + 2x + 8 = 8x + 12 = 4(2x + 3)$
\end{solution}
\end{parts}
