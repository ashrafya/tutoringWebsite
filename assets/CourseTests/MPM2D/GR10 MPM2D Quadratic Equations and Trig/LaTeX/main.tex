\question \textbf{Solving Quadratic Equations. (6 marks)}

\begin{parts}
\part[2] Solve by factoring: $x^2 - 3x - 18 = 0$

\begin{solution}[4cm]
\[
(x - 6)(x + 3) = 0
\]
\[
x = 6 \quad \text{or} \quad x = -3
\]
\end{solution}

\part[2] Solve by factoring: $2x^2 + 7x + 3 = 0$

\begin{solution}[5cm]
Factor: $(2x + 1)(x + 3) = 0$\\
$x = -\dfrac{1}{2}$ or $x = -3$
\end{solution}

\part[2] Solve using the quadratic formula: $3x^2 - 5x - 1 = 0$. Leave answers in exact form.

\begin{solution}[5cm]
$a = 3,\ b = -5,\ c = -1$.
\[
x = \frac{5 \pm \sqrt{25 + 12}}{6} = \frac{5 \pm \sqrt{37}}{6}
\]
\[
x = \frac{5 + \sqrt{37}}{6} \quad \text{or} \quad x = \frac{5 - \sqrt{37}}{6}
\]
\end{solution}
\end{parts}

\question \textbf{Discriminant and Nature of Roots. (4 marks)}\\
For each equation, calculate the discriminant and state the number and type of roots.

\begin{parts}
\part[1] $x^2 - 6x + 9 = 0$
\begin{solution}[2.5cm]
$\Delta = 36 - 36 = 0$. \textbf{One repeated real root.}
\end{solution}

\part[1] $2x^2 + 3x - 5 = 0$
\begin{solution}[2.5cm]
$\Delta = 9 + 40 = 49 > 0$. \textbf{Two distinct real roots.}
\end{solution}

\part[2] $x^2 + x + 4 = 0$
\begin{solution}[2.5cm]
$\Delta = 1 - 16 = -15 < 0$. \textbf{No real roots} (two complex roots).
\end{solution}
\end{parts}

\question \textbf{Trigonometry. (7 marks)}

\begin{parts}
\part[2] In right triangle $PQR$, $\angle R = 90°$, $PQ = 17$, and $\angle P = 35°$. Find the length of $QR$ to the nearest tenth.

\begin{solution}[5cm]
$QR$ is opposite to $\angle P$, $PQ$ is the hypotenuse.
\[
\sin 35° = \frac{QR}{17} \implies QR = 17\sin 35° \approx 17 \times 0.5736 \approx 9.8
\]
\end{solution}

\part[2] In right triangle $ABC$, $\angle C = 90°$, $AB = 15$~cm, and $BC = 9$~cm. Find $\angle A$ to the nearest degree.

\begin{solution}[5cm]
$BC$ is adjacent to $\angle A$, $AB$ is the hypotenuse.
\[
\cos A = \frac{BC}{AB} = \frac{9}{15} = 0.6 \implies \angle A = \cos^{-1}(0.6) \approx 53°
\]
\end{solution}

\part[1] Find the missing side in a right triangle where $\angle \theta = 45°$ and the adjacent side is 8 cm. Find the opposite side.

\begin{solution}[3cm]
$\tan 45° = 1$, so opposite $= 8 \times \tan 45° = 8 \times 1 = 8$ cm.
\end{solution}

\part[2] Without a calculator, state the exact values of $\sin 30°$, $\cos 60°$, and $\tan 45°$.

\begin{solution}[3cm]
$\sin 30° = \dfrac{1}{2}$, \quad $\cos 60° = \dfrac{1}{2}$, \quad $\tan 45° = 1$.
\end{solution}
\end{parts}

\question \textbf{Angles of Elevation and Depression. (8 marks)}

\begin{parts}
\part[4] A person standing 45 m from the base of a flagpole measures the angle of elevation to the top as $62°$. How tall is the flagpole? Round to the nearest tenth of a metre.

\begin{solution}[7cm]
Let $h$ = height of flagpole.
\[
\tan 62° = \frac{h}{45}
\]
\[
h = 45 \tan 62° \approx 45 \times 1.8807 \approx 84.6 \text{ m}
\]
\end{solution}

\part[4] A lighthouse keeper at the top of a 60 m tall lighthouse spots a ship at sea. The angle of depression to the ship is $18°$. How far is the ship from the base of the lighthouse? Round to the nearest metre.

\begin{solution}[7cm]
The angle of depression from the top of the lighthouse equals the angle of elevation from the ship to the top (alternate interior angles).

Let $d$ = horizontal distance.
\[
\tan 18° = \frac{60}{d}
\]
\[
d = \frac{60}{\tan 18°} \approx \frac{60}{0.3249} \approx 185 \text{ m}
\]
\end{solution}
\end{parts}

\question \textbf{Two-Step Trigonometry Problem. (8 marks)}\\
A surveyor wants to find the height of a building. From point $A$, she measures the angle of elevation to the top of the building as $50°$. She then walks 30 m closer to the building to point $B$ and measures the angle of elevation as $70°$. Find the height of the building to the nearest tenth of a metre.

\begin{solution}[12cm]
Let $h$ = height of building and $d$ = horizontal distance from $B$ to the base of the building.

From point $B$:
\[
\tan 70° = \frac{h}{d} \implies h = d\tan 70°  \quad \cdots (1)
\]
From point $A$ (which is $d + 30$ m from the base):
\[
\tan 50° = \frac{h}{d + 30} \implies h = (d + 30)\tan 50°  \quad \cdots (2)
\]
Set (1) $=$ (2):
\[
d\tan 70° = (d + 30)\tan 50°
\]
\[
d\tan 70° = d\tan 50° + 30\tan 50°
\]
\[
d(\tan 70° - \tan 50°) = 30\tan 50°
\]
\[
d = \frac{30\tan 50°}{\tan 70° - \tan 50°} \approx \frac{30 \times 1.1918}{2.7475 - 1.1918} = \frac{35.754}{1.5557} \approx 22.98 \text{ m}
\]
\[
h = d\tan 70° \approx 22.98 \times 2.7475 \approx 63.2 \text{ m}
\]
The building is approximately \textbf{63.2 m} tall.
\end{solution}
