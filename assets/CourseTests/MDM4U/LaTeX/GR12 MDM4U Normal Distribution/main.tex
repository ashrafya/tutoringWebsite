\section*{Part B --- Short Answer (33 marks)}

\question The lifespans of a certain brand of light bulb are normally distributed with
mean $\mu = 1200$ hours and standard deviation $\sigma = 100$ hours.
\begin{parts}
    \part[2] What proportion of bulbs last between 1100 and 1350 hours?
    \part[2] What proportion of bulbs last \textbf{less than} 1050 hours?
    \part[2] What proportion of bulbs last \textbf{more than} 1420 hours?
    \part[2] Between what two lifespans do the middle 90\% of bulbs fall?
\end{parts}
\begin{solution}[10cm]
\textbf{(a)} $z_1 = \dfrac{1100-1200}{100} = -1.0$;\quad $z_2 = \dfrac{1350-1200}{100} = 1.5$
\[
P(-1<Z<1.5) = P(Z<1.5) - P(Z<-1) = 0.9332 - 0.1587 = \mathbf{0.7745}
\]

\textbf{(b)} $z = \dfrac{1050-1200}{100} = -1.5$
\[
P(Z < -1.5) = \mathbf{0.0668}
\]

\textbf{(c)} $z = \dfrac{1420-1200}{100} = 2.2$
\[
P(Z > 2.2) = 1 - P(Z < 2.2) = 1 - 0.9861 = \mathbf{0.0139}
\]

\textbf{(d)} Middle 90\% $\Rightarrow$ 5\% in each tail $\Rightarrow z = \pm 1.645$.
\[
x = \mu \pm z\sigma = 1200 \pm 1.645(100) \implies \mathbf{[1035.5,\; 1364.5] \text{ hours}}
\]
\end{solution}

\question A university entrance exam has scores normally distributed with $\mu = 500$ and
$\sigma = 80$.
\begin{parts}
    \part[2] The top 10\% of students are offered scholarships. What is the minimum score
    needed for a scholarship?
    \part[2] A student scores 620. What percentile does this correspond to?
    \part[3] If the university accepts students who score between 440 and 620, what
    percentage of students are accepted?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} Top 10\%: $P(Z > z) = 0.10 \Rightarrow P(Z < z) = 0.90 \Rightarrow z = 1.28$.
\[
x = 500 + 1.28(80) = 500 + 102.4 = \mathbf{602.4 \approx 602}
\]

\textbf{(b)} $z = \dfrac{620-500}{80} = 1.50$.
\[
P(Z < 1.50) = 0.9332 \implies \textbf{93rd percentile}
\]

\textbf{(c)} $z_1 = \dfrac{440-500}{80} = -0.75$;\quad $z_2 = \dfrac{620-500}{80} = 1.50$
\[
P(-0.75 < Z < 1.50) = P(Z<1.50) - P(Z<-0.75) = 0.9332 - 0.2266 = \mathbf{0.7066 \approx 70.7\%}
\]
\end{solution}

\question The mass of apples from an orchard is normally distributed with mean 182 g and
standard deviation 24 g.
\begin{parts}
    \part[2] A store rejects apples under 150 g or over 230 g. What proportion of the
    harvest is rejected?
    \part[2] Find the mass such that 75\% of apples are lighter.
    \part[3] In a crate of 500 apples, how many are expected to weigh more than 200 g?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $z_1 = \dfrac{150-182}{24} = -1.33$;\quad $z_2 = \dfrac{230-182}{24} = 2.0$
\[
P(X<150) = P(Z<-1.33) = 0.0918;\quad P(X>230) = P(Z>2.0) = 0.0228
\]
\[
P(\text{rejected}) = 0.0918 + 0.0228 = \mathbf{0.1146 \approx 11.5\%}
\]

\textbf{(b)} 75th percentile: $z = 0.674$.
\[
x = 182 + 0.674(24) = 182 + 16.18 \approx \mathbf{198.2 \text{ g}}
\]

\textbf{(c)} $z = \dfrac{200-182}{24} = 0.75$;\quad $P(X>200) = 1 - P(Z<0.75) = 1 - 0.7734 = 0.2266$
\[
\text{Expected count} = 500 \times 0.2266 \approx \mathbf{113 \text{ apples}}
\]
\end{solution}

\question A factory produces bolts with mean diameter 12.00 mm and standard deviation
0.08 mm. A sample of $n = 64$ bolts is taken.
\begin{parts}
    \part[2] State the distribution of the sample mean $\bar{X}$, giving the mean and
    standard error.
    \part[3] Find the probability that the sample mean diameter is between 11.98 mm
    and 12.02 mm.
    \part[2] How large a sample would be needed to reduce the standard error to 0.005 mm?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} By the Central Limit Theorem:
$\bar{X} \sim N\!\left(12.00,\;\left(\dfrac{0.08}{\sqrt{64}}\right)^2\right)
= N(12.00,\; 0.01^2)$

Standard error $= \dfrac{0.08}{\sqrt{64}} = \dfrac{0.08}{8} = \mathbf{0.01}$ mm.

\textbf{(b)} $z_1 = \dfrac{11.98 - 12.00}{0.01} = -2.0$;\quad
$z_2 = \dfrac{12.02 - 12.00}{0.01} = 2.0$
\[
P(-2 < Z < 2) = 0.9544 - 0.0456 = \mathbf{0.9544}
\]

\textbf{(c)} SE $= \dfrac{\sigma}{\sqrt{n}} = 0.005 \Rightarrow \sqrt{n} = \dfrac{0.08}{0.005} = 16
\Rightarrow n = \mathbf{256}$
\end{solution}
