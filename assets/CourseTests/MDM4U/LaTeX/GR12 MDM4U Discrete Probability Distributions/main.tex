\question[1] A discrete random variable $X$ can take values $\{1, 2, 3\}$ with probabilities
$0.2, 0.5, 0.3$. Which statement is true?
\begin{choices}
    \choice The distribution is invalid because probabilities are not equal.
    \correctchoice The distribution is valid because the probabilities sum to 1.
    \choice $P(X=2)$ must equal $0.333$.
    \choice $X$ is a continuous variable.
\end{choices}

\section*{Part B --- Short Answer (33 marks)}

\question A game is played by spinning a wheel that lands on \$0, \$2, \$5, or \$10
with probabilities 0.40, 0.30, 0.20, and 0.10 respectively. It costs \$2 to play.
\begin{parts}
    \part[2] Construct the probability distribution table for winnings $W$ (prize minus cost).
    \part[3] Calculate $E[W]$ and $\text{Var}(W)$.
    \part[2] Is the game fair? What is the expected gain/loss per game?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $W = \text{prize} - 2$:

\begin{center}
\begin{tabular}{ccccc}
\toprule
$w$ & $-2$ & $0$ & $3$ & $8$ \\
\midrule
$P(W=w)$ & $0.40$ & $0.30$ & $0.20$ & $0.10$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{(b)}
\[
E[W] = (-2)(0.40) + (0)(0.30) + (3)(0.20) + (8)(0.10)
= -0.8 + 0 + 0.6 + 0.8 = \mathbf{0.60}
\]
\[
E[W^2] = 4(0.40)+0(0.30)+9(0.20)+64(0.10)
= 1.6+0+1.8+6.4 = 9.8
\]
\[
\text{Var}(W) = E[W^2] - (E[W])^2 = 9.8 - 0.36 = \mathbf{9.44}
\]

\textbf{(c)} $E[W] = \$0.60 > 0$, so the game \textbf{favours the player} (expected gain of 60 cents per game).
\end{solution}

\question A multiple-choice quiz has 10 questions, each with 4 choices (exactly one correct).
A student guesses randomly on every question. Let $X$ = number of correct answers.
\begin{parts}
    \part[1] State the distribution of $X$ with its parameters.
    \part[2] Find $P(X = 4)$.
    \part[2] Find $P(X \geq 2)$.
    \part[2] Find the mean and standard deviation of $X$.
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $X \sim B\!\left(10,\,\dfrac{1}{4}\right)$

\textbf{(b)}
\[
P(X=4) = \binom{10}{4}\!\left(\tfrac{1}{4}\right)^4\!\left(\tfrac{3}{4}\right)^6
= 210 \times \frac{1}{256} \times \frac{729}{4096}
= \frac{210 \times 729}{1\,048\,576} \approx \mathbf{0.1460}
\]

\textbf{(c)} Use complement:
\[
P(X \geq 2) = 1 - P(X=0) - P(X=1)
\]
\[
P(X=0) = \left(\tfrac{3}{4}\right)^{10} \approx 0.0563;\quad
P(X=1) = 10\left(\tfrac{1}{4}\right)\!\left(\tfrac{3}{4}\right)^9 \approx 0.1877
\]
\[
P(X \geq 2) \approx 1 - 0.0563 - 0.1877 = \mathbf{0.7560}
\]

\textbf{(d)} $\mu = np = 10 \times 0.25 = \mathbf{2.5}$;
$\sigma = \sqrt{np(1-p)} = \sqrt{10 \times 0.25 \times 0.75} = \sqrt{1.875} \approx \mathbf{1.369}$
\end{solution}

\question A committee of 4 is chosen at random from a group of 6 economists and 4 engineers.
Let $X$ = the number of engineers on the committee.
\begin{parts}
    \part[1] State the distribution of $X$ with all parameters.
    \part[3] Find $P(X = 2)$.
    \part[2] Find $P(X \geq 3)$.
    \part[2] Find $E[X]$.
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} Hypergeometric: $N=10$, $K=4$ engineers, $n=4$ drawn.
\[
P(X=k) = \frac{\dbinom{4}{k}\dbinom{6}{4-k}}{\dbinom{10}{4}}
\]

\textbf{(b)}
\[
P(X=2) = \frac{\dbinom{4}{2}\dbinom{6}{2}}{\dbinom{10}{4}}
= \frac{6 \times 15}{210} = \frac{90}{210} = \mathbf{\frac{3}{7} \approx 0.4286}
\]

\textbf{(c)}
\[
P(X=3) = \frac{\dbinom{4}{3}\dbinom{6}{1}}{\dbinom{10}{4}} = \frac{4 \times 6}{210} = \frac{24}{210}
\]
\[
P(X=4) = \frac{\dbinom{4}{4}\dbinom{6}{0}}{\dbinom{10}{4}} = \frac{1}{210}
\]
\[
P(X \geq 3) = \frac{24+1}{210} = \frac{25}{210} = \mathbf{\frac{5}{42} \approx 0.1190}
\]

\textbf{(d)} $E[X] = n \cdot \dfrac{K}{N} = 4 \times \dfrac{4}{10} = \mathbf{1.6}$
\end{solution}

\question A basketball player makes free throws with probability 0.70. She shoots until
she makes her first basket.
\begin{parts}
    \part[1] State the distribution of $X$ (number of shots until first basket) with its
    parameter.
    \part[2] Find the probability she makes her first basket on the 3rd shot.
    \part[2] Find the probability she needs \textbf{more than 3} shots.
    \part[2] Find the expected number of shots needed.
\end{parts}
\begin{solution}[8cm]
\textbf{(a)} $X \sim \text{Geometric}(p = 0.70)$

\textbf{(b)}
\[
P(X=3) = (0.30)^2(0.70) = 0.09 \times 0.70 = \mathbf{0.063}
\]

\textbf{(c)}
\[
P(X > 3) = (1-p)^3 = (0.30)^3 = \mathbf{0.027}
\]

\textbf{(d)} $E[X] = \dfrac{1}{p} = \dfrac{1}{0.70} \approx \mathbf{1.43 \text{ shots}}$
\end{solution}
