\section*{Part B --- Short Answer (40 marks)}

\question \textbf{(Counting)} A student council of 5 is to be chosen from 8 boys and 6 girls.
\begin{parts}
    \part[2] How many councils are possible with no restrictions?
    \part[3] How many councils have exactly 3 boys and 2 girls?
    \part[3] How many councils have at least 4 girls?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $\dbinom{14}{5} = \mathbf{2002}$

\textbf{(b)} $\dbinom{8}{3}\dbinom{6}{2} = 56 \times 15 = \mathbf{840}$

\textbf{(c)} (exactly 4 girls) + (exactly 5 girls):
\[
\binom{6}{4}\binom{8}{1} + \binom{6}{5}\binom{8}{0}
= 15 \times 8 + 6 \times 1 = 120 + 6 = \mathbf{126}
\]
\end{solution}

\question \textbf{(Probability)} A company finds that 3\% of their products are defective.
A quality control inspector tests items one by one until finding a defective item.
\begin{parts}
    \part[2] Find the probability the first defective item is the 5th item tested.
    \part[2] Find the probability the first defective item is found within the first
    3 tests.
    \part[2] Find the expected number of items tested until the first defect is found.
\end{parts}
\begin{solution}[8cm]
Let $p = 0.03$, $q = 0.97$.

\textbf{(a)} $P(X=5) = (0.97)^4(0.03) = 0.885^4 \times 0.03$.
$(0.97)^4 = 0.8853$, so $P(X=5) = 0.8853 \times 0.03 \approx \mathbf{0.0266}$

\textbf{(b)}
\[
P(X \leq 3) = P(X=1)+P(X=2)+P(X=3)
= 0.03 + (0.97)(0.03) + (0.97)^2(0.03)
= 0.03(1 + 0.97 + 0.9409) \approx \mathbf{0.0873}
\]

\textbf{(c)} $E[X] = \dfrac{1}{p} = \dfrac{1}{0.03} \approx \mathbf{33.3 \text{ items}}$
\end{solution}

\question \textbf{(Discrete Distributions)} A carnival game involves drawing 3 cards from
a deck of 10 cards numbered 1--10. A player wins \$5 for each even-numbered card drawn.
Let $X$ = number of even cards drawn.
\begin{parts}
    \part[1] Identify the distribution of $X$ and state its parameters.
    \part[3] Construct the complete probability distribution table for $X$.
    \part[3] Calculate $E[X]$ and interpret in context.
\end{parts}
\begin{solution}[10cm]
\textbf{(a)} Hypergeometric: $N=10$, $K=5$ even cards $\{2,4,6,8,10\}$, $n=3$ drawn.

\textbf{(b)} $\dbinom{10}{3} = 120$.
\[
P(X=0) = \frac{\binom{5}{0}\binom{5}{3}}{120} = \frac{10}{120} = \frac{1}{12}
\]
\[
P(X=1) = \frac{\binom{5}{1}\binom{5}{2}}{120} = \frac{5\times10}{120} = \frac{50}{120} = \frac{5}{12}
\]
\[
P(X=2) = \frac{\binom{5}{2}\binom{5}{1}}{120} = \frac{10\times5}{120} = \frac{50}{120} = \frac{5}{12}
\]
\[
P(X=3) = \frac{\binom{5}{3}\binom{5}{0}}{120} = \frac{10}{120} = \frac{1}{12}
\]

\textbf{(c)}
$E[X] = 0\cdot\dfrac{1}{12} + 1\cdot\dfrac{5}{12} + 2\cdot\dfrac{5}{12} + 3\cdot\dfrac{1}{12}
= \dfrac{0+5+10+3}{12} = \dfrac{18}{12} = \mathbf{1.5}$

On average, a player draws 1.5 even cards per game, winning $\$5 \times 1.5 = \$7.50$.
\end{solution}

\question \textbf{(Statistics)} The following data shows the monthly sales (in thousands of
dollars) for a small business over 10 months:
\[42,\; 47,\; 51,\; 55,\; 58,\; 62,\; 65,\; 70,\; 74,\; 96\]
\begin{parts}
    \part[3] Find the mean, median, and standard deviation.
    \part[2] Find the five-number summary and identify any outliers using the $1.5 \times \text{IQR}$ rule.
    \part[3] The owner projects that next year's monthly sales will each increase by 10\%
    (multiply by 1.1). State the new mean and standard deviation without recalculating
    from scratch.
\end{parts}
\begin{solution}[10cm]
\textbf{(a)} $\bar{x} = \dfrac{42+47+51+55+58+62+65+70+74+96}{10} = \dfrac{620}{10} = \mathbf{62}$

Median $= \dfrac{58+62}{2} = \mathbf{60}$

Deviations from 62: $-20,-15,-11,-7,-4,0,3,8,12,34$.
$\sum d^2 = 400+225+121+49+16+0+9+64+144+1156 = 2184$
$\sigma = \sqrt{2184/10} = \sqrt{218.4} \approx \mathbf{14.78}$

\textbf{(b)} $Q_1 = \dfrac{51+55}{2} = 53$;\; $Q_3 = \dfrac{70+74}{2} = 72$;\; IQR $= 19$\\
Upper fence: $72 + 1.5(19) = 72 + 28.5 = 100.5$. All values $\leq 96 < 100.5$.\\
\textbf{No outliers.} Five-number summary: $\{42,\; 53,\; 60,\; 72,\; 96\}$.

\textbf{(c)} Multiplying all values by 1.1:
New mean $= 62 \times 1.1 = \mathbf{\$68.2\text{k}}$;\quad
New $\sigma = 14.78 \times 1.1 = \mathbf{\$16.26\text{k}}$.
\end{solution}

\question \textbf{(Normal Distribution)} The time it takes students to complete a
standardized test is normally distributed with mean 95 minutes and standard deviation
12 minutes.
\begin{parts}
    \part[2] What proportion of students finish in under 80 minutes?
    \part[2] What proportion finish between 90 and 110 minutes?
    \part[2] The fastest 15\% of students get a time bonus. What is the maximum time
    a student can take and still receive the bonus?
    \part[2] In a class of 30 students, how many are expected to take more than 115
    minutes?
\end{parts}
\begin{solution}[9cm]
\textbf{(a)} $z = \dfrac{80-95}{12} = -1.25$;\quad $P(Z<-1.25) = 1 - 0.8944 = \mathbf{0.1056}$

\textbf{(b)} $z_1 = \dfrac{90-95}{12} = -0.42$;\quad $z_2 = \dfrac{110-95}{12} = 1.25$
\[
P(-0.42 < Z < 1.25) = 0.8944 - 0.3372 = \mathbf{0.5572}
\]

\textbf{(c)} Fastest 15\%: $P(Z < z) = 0.15 \Rightarrow z = -1.04$
\[
x = 95 + (-1.04)(12) = 95 - 12.48 = \mathbf{82.52 \text{ minutes}}
\]

\textbf{(d)} $z = \dfrac{115-95}{12} = 1.67$;\quad $P(Z>1.67) = 1 - 0.9525 = 0.0475$
\[
\text{Expected count} = 30 \times 0.0475 \approx \mathbf{1.4 \approx 1 \text{ student}}
\]
\end{solution}
